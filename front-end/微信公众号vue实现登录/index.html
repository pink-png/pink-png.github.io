微信公众号vue实现登录

发表于 2021-06-25
  • Yume Shoka
首页 前端开发 微信公众号

1. 在Vue项目中安装并引入微信 JS-SDK库（jweixin.js）。
npm install weixin-js-sdk
引入 import wx from 'weixin-js-sdk';

2. 在Vue的组件中，创建一个用于授权登录的按钮，并定义触发登录的方法。
<template>
  <div>
    <button @click="wechatLogin">微信登录</button>
  </div>
</template>

methods: {
  wechatLogin() {
    // 微信公众号授权登录的逻辑
  }
}

3. 在微信授权登录的方法内部，进行微信 JS-SDK的初始化和授权登录操作。
// 引入后端提供的获取微信 JS-SDK配置的接口
import { getWechatConfig } from '@/api/wechat'; // 替换成实际的接口路径

methods: {
  async wechatLogin() {
    const config = await getWechatConfig(); // 调用后端接口获取微信 JS-SDK的配置信息
    wx.config({
      debug: false,
      appId: config.appId,
      timestamp: config.timestamp,
      nonceStr: config.nonceStr,
      signature: config.signature,
      jsApiList: ['checkJsApi', 'authorize'] // 需要使用的微信 API 列表
    });

    wx.error(function(res) {
      // 初始化失败的处理逻辑
      console.error('微信 JS-SDK初始化失败:', res);
    });

    const vm = this;
    wx.ready(function() {
      // 在这里执行授权登录操作
      wx.authorize({
        scope: 'snsapi_base',
        success: function(res) {
          // 用户同意授权，可以获取到用户的基本信息
          const code = res.code; // 获取到授权登录的code
          // 在这里可以将code发送到服务器进行进一步处理
          console.log('授权登录成功，code:', code);
        },
        fail: function(res) {
          // 用户拒绝授权或授权失败
          console.log('授权登录失败:', res);
        }
      });
    });
  }
}
